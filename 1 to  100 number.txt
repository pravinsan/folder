
#include <stdio.h>

int main()
{ int x =1;
   while(x<101)
     {
         if(x<=10)
           {
               printf("%d",x);
           }
           else
           {
               printf("%d",x);
           }
           {
               if(x%10==0)
                 {
                     printf("\n");
                 }
                 else
                 {
                     printf(" ");
                 }
                 x++;
           }
     }
  

    return 0;
}
